database name---------->use meeting_room_booking;

*******register table**********

create table register(
id int not null auto_increment unique,
username varchar(45) primary key,
email varchar(45),
password varchar(45),
phone varchar(10),
occupancy varchar(200),
token varchar(200)
);

********meetingSchedule table*************

create table meetingSchedule(
id int auto_increment primary key,
attendees varchar(300),
title varchar(100),
loginEmail varchar(45) ,
startdate date,
enddate date,
starttime time,
endtime time,
roomname varchar(45)
); 


***********rooms table**************

create table rooms 
(id int primary key auto_increment,roomName varchar(50),occupancyString varchar(100));
